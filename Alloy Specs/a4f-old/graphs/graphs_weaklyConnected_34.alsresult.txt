sig Node {
    adj : set Node
}

pred undirected {
    adj = ~adj     // fixed
}

pred oriented {
    no (adj & ~adj)    // fixed
}

pred acyclic {
    all a:Node | a not in a.^adj    // fixed
}

pred complete {
    all n:Node | n.adj = Node - n    // fixed
}
